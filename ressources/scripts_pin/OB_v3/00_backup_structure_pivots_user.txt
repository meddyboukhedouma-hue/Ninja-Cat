//@version=6
// SAUVEGARDE de la version éditeur de l'utilisateur (détection PIVOTS, len=5) avant bascule cassage-de-structure.
// Structure de marché HH / HL / LH / LL + courbe ATR(SMA 233).
indicator("Structure HH/HL/LH/LL", overlay=false, max_lines_count=500, max_labels_count=500)

len     = input.int(5, "Longueur des pivots", minval=1, maxval=50, group="Structure")
lineCol = input.color(color.new(color.gray, 0),  "Zigzag",    group="Style")
colUp   = input.color(color.new(color.teal, 0),  "HH / HL",   group="Style")
colDn   = input.color(color.new(color.red, 0),   "LH / LL",   group="Style")
lineW   = input.int(2, "Épaisseur", minval=1, maxval=5, group="Style")

atrSmooth = input.string("SMA", "Lissage ATR", options=["RMA", "SMA", "EMA", "WMA"], group="ATR")
atrLen    = input.int(233, "Longueur ATR", minval=1, group="ATR")
atrColIn  = input.color(color.new(#B71C1C, 0), "Couleur ATR", group="ATR")

f_ma(float src, int l) =>
    switch atrSmooth
        "RMA" => ta.rma(src, l)
        "SMA" => ta.sma(src, l)
        "EMA" => ta.ema(src, l)
        => ta.wma(src, l)

atrV = f_ma(ta.tr(true), atrLen)
plot(atrV, "ATR", color=atrColIn, linewidth=2)

ph = ta.pivothigh(high, len, len)
pl = ta.pivotlow(low,  len, len)
pt = time[len]

var int   dir = 0
var float curPx = na
var int   curTime = na
var float anchorPx = na
var int   anchorTime = na
var float prevHigh = na
var float prevLow  = na
var float lastHigh = na
var float lastLow  = na
var line  segLine  = na
var label segLabel = na

f_seg(int t1, float p1, int t2, float p2) =>
    line.new(t1, p1, t2, p2, xloc=xloc.bar_time, color=lineCol, width=lineW, force_overlay=true)

if not na(ph)
    isNew = dir != 1
    isExt = dir == 1 and ph > curPx
    if isNew or isExt
        if isNew
            prevHigh   := lastHigh
            anchorPx   := curPx
            anchorTime := curTime
            dir        := 1
        if isExt
            if not na(segLine)
                line.delete(segLine)
            if not na(segLabel)
                label.delete(segLabel)
        curPx   := ph
        curTime := pt
        isHH = na(prevHigh) or ph > prevHigh
        cc   = isHH ? colUp : colDn
        if not na(anchorTime)
            segLine := f_seg(anchorTime, anchorPx, pt, ph)
        segLabel := label.new(pt, ph, isHH ? "HH" : "LH", xloc=xloc.bar_time, yloc=yloc.price, style=label.style_label_down, color=color.new(cc, 85), textcolor=cc, size=size.small, force_overlay=true)
        lastHigh := ph

if not na(pl)
    isNew = dir != -1
    isExt = dir == -1 and pl < curPx
    if isNew or isExt
        if isNew
            prevLow    := lastLow
            anchorPx   := curPx
            anchorTime := curTime
            dir        := -1
        if isExt
            if not na(segLine)
                line.delete(segLine)
            if not na(segLabel)
                label.delete(segLabel)
        curPx   := pl
        curTime := pt
        isHL = na(prevLow) or pl > prevLow
        cc   = isHL ? colUp : colDn
        if not na(anchorTime)
            segLine := f_seg(anchorTime, anchorPx, pt, pl)
        segLabel := label.new(pt, pl, isHL ? "HL" : "LL", xloc=xloc.bar_time, yloc=yloc.price, style=label.style_label_up, color=color.new(cc, 85), textcolor=cc, size=size.small, force_overlay=true)
        lastLow := pl
