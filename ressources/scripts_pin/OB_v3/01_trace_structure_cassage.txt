//@version=6
// Structure HH/HL/LH/LL — tracé piloté par les CASSURES.
// Les "points de structure" = pivots 2 bougies (granularité minimale viable) ; le niveau cassé ratchete.
indicator("Structure HH/HL/LH/LL — cassage", overlay=true, max_lines_count=500, max_labels_count=500)

lineCol = input.color(color.new(color.gray, 0),  "Zigzag",  group="Style")
colUp   = input.color(color.new(color.teal, 0),  "HH / HL", group="Style")
colDn   = input.color(color.new(color.red, 0),   "LH / LL", group="Style")
lineW   = input.int(2, "Épaisseur", minval=1, maxval=5, group="Style")

atrSmooth = input.string("SMA", "Lissage ATR", options=["RMA", "SMA", "EMA", "WMA"], group="ATR")
atrLen    = input.int(233, "Longueur ATR", minval=1, group="ATR")
atrColIn  = input.color(color.new(#B71C1C, 0), "Couleur ATR", group="ATR")

f_ma(float src, int l) =>
    switch atrSmooth
        "RMA" => ta.rma(src, l)
        "SMA" => ta.sma(src, l)
        "EMA" => ta.ema(src, l)
        => ta.wma(src, l)
atrV = f_ma(ta.tr(true), atrLen)
plot(atrV, "ATR", color=atrColIn, display=display.data_window)

ph = ta.pivothigh(high, 2, 2)
pl = ta.pivotlow(low,  2, 2)

var float upper     = na
var int   upperT    = na
var float lower     = na
var int   lowerT    = na
var int   trend     = 0
var float curExt    = na
var int   curExtT   = na
var float prevHigh  = na
var float prevLow   = na
var float lastPtPx  = na
var int   lastPtT   = na

f_seg(int t1, float p1, int t2, float p2) =>
    line.new(t1, p1, t2, p2, xloc=xloc.bar_time, color=lineCol, width=lineW)

if not na(ph)
    upper  := ph
    upperT := time[2]
if not na(pl)
    lower  := pl
    lowerT := time[2]

// ── Initialisation du trend ──
if trend == 0 and not na(upper) and not na(lower)
    if upperT > lowerT
        trend    := 1
        curExt   := high
        curExtT  := time
        prevLow  := lower
        lastPtPx := lower
        lastPtT  := lowerT
    else
        trend    := -1
        curExt   := low
        curExtT  := time
        prevHigh := upper
        lastPtPx := upper
        lastPtT  := upperT

// ── Tracking de l'extrême courant ──
if trend == 1
    if high > curExt
        curExt  := high
        curExtT := time
else if trend == -1
    if low < curExt
        curExt  := low
        curExtT := time

// ── Détection des cassures et basculements ──
if trend == 1 and not na(lower) and close < lower
    isHH = na(prevHigh) ? false : curExt > prevHigh
    cc   = isHH ? colUp : colDn
    if not na(lastPtT)
        f_seg(lastPtT, lastPtPx, curExtT, curExt)
    label.new(curExtT, curExt, isHH ? "HH" : "LH", xloc=xloc.bar_time, yloc=yloc.price,
              style=label.style_label_down, color=color.new(cc, 85), textcolor=cc, size=size.small)
    prevHigh := curExt
    lastPtPx := curExt
    lastPtT  := curExtT
    trend    := -1
    curExt   := low
    curExtT  := time

else if trend == -1 and not na(upper) and close > upper
    isHL = na(prevLow) ? false : curExt > prevLow
    cc   = isHL ? colUp : colDn
    if not na(lastPtT)
        f_seg(lastPtT, lastPtPx, curExtT, curExt)
    label.new(curExtT, curExt, isHL ? "HL" : "LL", xloc=xloc.bar_time, yloc=yloc.price,
              style=label.style_label_up, color=color.new(cc, 85), textcolor=cc, size=size.small)
    prevLow  := curExt
    lastPtPx := curExt
    lastPtT  := curExtT
    trend    := 1
    curExt   := high
    curExtT  := time
