//@version=6
// Structure HH/HL/LH/LL — pilotée par le CASSAGE DE STRUCTURE, points repérés par un petit fractal.
//   F (petit) = repère les sommets/creux LOCAUX candidats. Ce n'est PAS une « longueur de swing » :
//   le tracé n'est pas fait des pivots eux-mêmes, mais des CASSURES.
//   Jambe haussière : on suit le plus-haut ; elle se TERMINE quand close < dernier CREUX fractal
//     (niveau protégé qui RATCHET avec les nouveaux creux) → l'extrême de la jambe devient un SOMMET (HH/LH).
//   Jambe baissière : symétrique ; close > dernier SOMMET fractal → l'extrême devient un CREUX (HL/LL).
//   = mécanique SWING DEV (cassage de structure) resserrée, dans les deux sens.
//   overlay=false : sous-fenêtre = COURBE ATR ; structure forcée sur les prix (force_overlay).
indicator("Structure HH/HL/LH/LL — cassage", overlay=false, max_lines_count=500, max_labels_count=500)

// repérage local minimal d'un haut/bas (interne, non réglable) — le TRACÉ reste défini par les CASSURES
int F   = 2
lineCol = input.color(color.new(color.gray, 0),  "Zigzag",  group="Style")
colUp   = input.color(color.new(color.teal, 0),  "HH / HL", group="Style")
colDn   = input.color(color.new(color.red, 0),   "LH / LL", group="Style")
lineW   = input.int(2, "Épaisseur", minval=1, maxval=5, group="Style")

atrSmooth = input.string("SMA", "Lissage ATR", options=["RMA", "SMA", "EMA", "WMA"], group="ATR")
atrLen    = input.int(233, "Longueur ATR", minval=1, group="ATR")
atrColIn  = input.color(color.new(#B71C1C, 0), "Couleur ATR", group="ATR")

f_ma(float src, int l) =>
    switch atrSmooth
        "RMA" => ta.rma(src, l)
        "SMA" => ta.sma(src, l)
        "EMA" => ta.ema(src, l)
        => ta.wma(src, l)
atrV = f_ma(ta.tr(true), atrLen)
plot(atrV, "ATR", color=atrColIn, linewidth=2)

// ════════ STRUCTURE PAR CASSAGE (fractal + ratchet) ════════
ph = ta.pivothigh(high, F, F)
pl = ta.pivotlow(low,  F, F)

var float upper     = na   // dernier SOMMET fractal (cassé pour finir une jambe baissière)
var int   upperT    = na
var float lower     = na   // dernier CREUX fractal (cassé pour finir une jambe haussière)
var int   lowerT    = na

var int   trend     = 0    // 1 = jambe haussière, -1 = baissière, 0 = amorçage
var float curExt    = na   // extrême de la jambe en cours
var int   curExtT   = na
var float prevHigh  = na   // dernier SOMMET confirmé (réf HH/LH)
var float prevLow   = na   // dernier CREUX confirmé (réf HL/LL)
var float lastPtPx  = na   // dernier point de structure (ancre du segment)
var int   lastPtT   = na

f_seg(int t1, float p1, int t2, float p2) =>
    line.new(t1, p1, t2, p2, xloc=xloc.bar_time, color=lineCol, width=lineW, force_overlay=true)

// niveaux fractals (ratchet)
if not na(ph)
    upper  := ph
    upperT := time[F]
if not na(pl)
    lower  := pl
    lowerT := time[F]

// amorçage : tendance = le fractal le plus récent
if trend == 0 and not na(upper) and not na(lower)
    if upperT > lowerT
        trend    := 1
        curExt   := high
        curExtT  := time
        prevLow  := lower
        lastPtPx := lower
        lastPtT  := lowerT
    else
        trend    := -1
        curExt   := low
        curExtT  := time
        prevHigh := upper
        lastPtPx := upper
        lastPtT  := upperT

// extension de la jambe en cours
if trend == 1 and high > curExt
    curExt  := high
    curExtT := time
if trend == -1 and low < curExt
    curExt  := low
    curExtT := time

// fin de jambe HAUSSIÈRE → close sous le dernier creux fractal → confirme un SOMMET
if trend == 1 and not na(lower) and close < lower
    isHH = na(prevHigh) or curExt > prevHigh
    cc   = isHH ? colUp : colDn
    if not na(lastPtT)
        f_seg(lastPtT, lastPtPx, curExtT, curExt)
    label.new(curExtT, curExt, isHH ? "HH" : "LH", xloc=xloc.bar_time, yloc=yloc.price, style=label.style_label_down, color=color.new(cc, 85), textcolor=cc, size=size.small, force_overlay=true)
    prevHigh := curExt
    lastPtPx := curExt
    lastPtT  := curExtT
    trend    := -1
    curExt   := low
    curExtT  := time

// fin de jambe BAISSIÈRE → close au-dessus du dernier sommet fractal → confirme un CREUX
if trend == -1 and not na(upper) and close > upper
    isHL = na(prevLow) or curExt > prevLow
    cc   = isHL ? colUp : colDn
    if not na(lastPtT)
        f_seg(lastPtT, lastPtPx, curExtT, curExt)
    label.new(curExtT, curExt, isHL ? "HL" : "LL", xloc=xloc.bar_time, yloc=yloc.price, style=label.style_label_up, color=color.new(cc, 85), textcolor=cc, size=size.small, force_overlay=true)
    prevLow  := curExt
    lastPtPx := curExt
    lastPtT  := curExtT
    trend    := 1
    curExt   := high
    curExtT  := time
