// purge ALIVE-AWARE : ne retire QUE le plus ancien OB MORT ; garde TOUS les OB actifs (vivants), quel que soit leur âge. maxOB = cible de purge des morts, jamais des vivants.
f_purge(OB[] arr) =>
    bool keepPurging = true
    while keepPurging and array.size(arr) > maxOB
        int idxDel = -1
        for ii = 0 to array.size(arr) - 1
            if idxDel < 0 and array.get(arr, ii).dead
                idxDel := ii
        if idxDel < 0
            keepPurging := false
        if idxDel >= 0
            old = array.remove(arr, idxDel)
            box.delete(old.bx)
            label.delete(old.lb)

        if iLo <= iHi

            bool twoOpp = false

            for idx = iLo to iHi
                int d = array.get(cDir, idx)

                    oppRun := oppRun + 1
                    if oppRun >= 2
                        twoOpp := true
                if rg >= minATR and dirMatch
                    has3x := true
            ok := has3x or (cnt >= 5 and oppTot <= 1 and not twoOpp)
    ok
